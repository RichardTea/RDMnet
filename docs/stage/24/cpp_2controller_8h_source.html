338 inline etcpal::Result Controller::Startup(ControllerNotifyHandler& notify_handler, const ControllerData& data,
339  const ControllerRdmData& rdm_data)
340 {
341  if (!rdm_data.IsValid())
342  return kEtcPalErrInvalid;
343 
344  notify_ = &notify_handler;
345  my_data_ = data;
346  my_rdm_data_ = rdm_data;
347 
348  RdmnetControllerConfig config;
349  rdmnet_controller_config_init(&config, data.uid.manufacturer_id());
350  config.optional.uid = data.uid.get();
351  config.optional.search_domain = data.search_domain.c_str();
352  config.cid = data.cid.get();
353  RDMNET_CONTROLLER_SET_CALLBACKS(&config, ControllerLibCbConnected, ControllerLibCbConnectFailed,
354  ControllerLibCbDisconnected, ControllerLibCbClientListUpdate,
355  ControllerLibCbRdmResponseReceived, ControllerLibCbStatusReceived, this);
356  RDMNET_CONTROLLER_SET_RDM_DATA(&config, rdm_data.manufacturer_label.c_str(),
357  rdm_data.device_model_description.c_str(), rdm_data.software_version_label.c_str(),
358  rdm_data.device_label.c_str());
359  return rdmnet_controller_create(&config, &handle_);
360 }

388  ControllerLibCbRdmResponseReceived, ControllerLibCbStatusReceived, this);
389  RDMNET_CONTROLLER_SET_RDM_CMD_CALLBACKS(&config, ControllerLibCbRdmCommandReceived,
