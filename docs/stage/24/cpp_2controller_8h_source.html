282 inline etcpal::Result Controller::Startup(ControllerNotifyHandler& notify_handler, const ControllerRdmData& rdm_data,
283  const ControllerData& data)
284 {
285  if (!rdm_data.IsValid())
286  return kEtcPalErrInvalid;
287 
288  notify_ = &notify_handler;
289  my_data_ = data;
290  my_rdm_data_ = rdm_data;
291 
292  RdmnetControllerConfig config;
293  rdmnet_controller_config_init(&config, data.uid.manufacturer_id);
294  config.optional.uid = data.uid.get();
295  config.optional.search_domain = data.search_domain.c_str();
296  config.cid = data.cid.get();
297  RDMNET_CONTROLLER_SET_CALLBACKS(&config, ControllerLibCbConnected, ControllerLibCbConnectFailed,
298  ControllerLibCbDisconnected, ControllerLibCbClientListUpdate,
299  ControllerLibCbRdmResponseReceived, ControllerLibCbStatusReceived, this);
300  RDMNET_CONTROLLER_SET_RDM_DATA(&config, rdm_data.manufacturer_label.c_str(),
301  rdm_data.device_model_description.c_str(), rdm_data.software_version_label.c_str(),
302  rdm_data.device_label.c_str());
303  return rdmnet_controller_create(&config, &handle_);
304 }
