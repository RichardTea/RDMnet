140 inline bool BrokerSettings::valid() const
141 {
142  // clang-format off
143  return (
144  !cid.IsNull() &&
145  (!scope.empty() && scope.length() < (E133_SCOPE_STRING_PADDED_LENGTH - 1)) &&
146  (!dns.manufacturer.empty() && dns.manufacturer.length() < (E133_MANUFACTURER_STRING_PADDED_LENGTH - 1)) &&
147  (!dns.model.empty() && dns.model.length() < (E133_MODEL_STRING_PADDED_LENGTH - 1)) &&
148  (!dns.service_instance_name.empty() && dns.service_instance_name.length() < (E133_SERVICE_NAME_STRING_PADDED_LENGTH - 1)) &&
149  (listen_port == 0 || listen_port >= 1024) &&
150  (RDM_GET_MANUFACTURER_ID(&uid) > 0 && RDM_GET_MANUFACTURER_ID(&uid) < 0x8000) &&
151  !(uid_type == rdmnet::BrokerSettings::kStaticUid && !RDMNET_UID_IS_STATIC(&uid)) &&
152  !(uid_type == rdmnet::BrokerSettings::kDynamicUid && !RDMNET_UID_IS_DYNAMIC(&uid))
153  );
154  // clang-format on
155 }
