45 // The interface for the listener callback.
46 class IListenThread_Notify
47 {
48 public:
49  // Called when the listen thread gets a new connection. If you return false,
50  // the connection is severed. Presumably you were just about to stop the
51  // listener threads anyway...
52  // The address & port fields of addr are used.
53  // Do NOT stop the listening thread in this callback!
54  virtual bool NewConnection(lwpa_socket_t new_sock, const LwpaSockaddr &remote_addr) = 0;
55 
56  // Called to log an error. You may want to stop the listening thread if
57  // errors keep occurring, but you should NOT do it in this callback!
58  virtual void LogError(const std::string &err) = 0;
59 };
